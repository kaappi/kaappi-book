% cover.tex - Full-bleed paperback wrap cover for
% "Kaappi: A Scheme Programming Language".
%
% Design: "Kaappi" is coffee (Malayalam/Tamil), so the cover leans into that.
% A warm espresso background carries a coffee cup whose steam rises into the
% Scheme lambda -- coffee and lambda calculus in one mark. Cream/caramel
% typography, a coffee-bean ornament, and a faint coffee-ring stain on the
% back keep it professional.
%
% The cup illustration (images/cover-cup.png, 2048x2048 RGBA, ~379 DPI at its
% printed size) was generated with the Gemini API (gemini-3-pro-image) and
% post-processed with a cosine alpha fade at the borders so it blends
% seamlessly into the espresso background. To regenerate it, run
% scripts/generate-cover-cup.py (needs GEMINI_API_KEY). Everything else on
% the cover is vector (TikZ).
%
% Built to KDP spec for a 6" x 9" black-&-white paperback. All horizontal
% geometry is parameterized on \PageCount (see the preamble block below):
%   Spine ....................... \PageCount x 0.002252"  (243pp -> 0.547")
%   Full cover (with bleed) ..... 12.25" + spine  x  9.250"  (243pp -> 12.797")
%   Trim (front / back) ......... 6.000" x 9.000"
%   Bleed ....................... 0.125" on every outer edge
%   Interior safety margin ...... 0.125" (we keep text >= 0.375")
%   Barcode keep-clear .......... 2.000" x 1.200", lower back cover
%
% When the interior page count changes: update \PageCount, rebuild, and
% eyeball `make cover-proof`. For a quick what-if proof without editing:
%   make cover-proof PAGES=400
%
% Compile with XeLaTeX (fontspec needs it):   make cover
% KDP guide overlay for proofing:             make cover-proof
% Output: build/cover.pdf, one page at the exact KDP sheet size.
%
% Layout on the flat sheet, left -> right:  BACK COVER | SPINE | FRONT COVER
% (KDP "Left to Right" reading order.)
%
% TikZ gotchas learned the hard way (keep in mind when editing):
%   - Nodes containing \\ line breaks MUST set text width, or every line
%     after the first silently reverts to the document's default font.
%   - pgfmath overflows above ~16384, so never compute i*angle in a loop;
%     accumulate the angle mod 360 instead.
%   - remember picture needs two passes; latexmk handles that (make cover),
%     the proof target runs XeLaTeX twice explicitly.

\documentclass[10pt]{article}

% --- Book parameters: the page count drives all horizontal geometry -----------
% KDP spine width = page count x per-page paper thickness:
%   0.002252 in/page  black & white on white paper   <- this book
%   0.0025   in/page  black & white on cream paper
%   0.002347 in/page  standard color
% Everything downstream (sheet width, spine folds, panel centres, proofing
% guides) derives from these two numbers. \def\PAGES{N} on the command line
% overrides \PageCount for what-if proofs (see the Makefile's cover-proof).
\newcommand{\SpinePerPage}{0.002252}
\ifdefined\PAGES
  \newcommand{\PageCount}{\PAGES}
\else
  \newcommand{\PageCount}{329}
\fi
% Multiply in exact decimal (\fpeval) BEFORE attaching the unit: TeX rounds a
% dimension's fraction to 1/65536 before unit scaling, so "0.002252in" is
% really 0.0022583in and the error would grow with every page (243 pages
% -> 0.0016" too wide). \SpineInches is the exact decimal, e.g. "0.547236".
\newcommand{\SpineInches}{\fpeval{\PageCount*\SpinePerPage}}
\newlength{\SpineWidth}
\setlength{\SpineWidth}{\SpineInches in}
% KDP allows spine text only on 80+ page books, and wants 0.0625" clearance
% between spine type and each fold. The 13pt spine type is ~0.18" tall, so
% below ~0.32" of spine it must shrink (or go entirely for very thin books).
\ifdim\SpineWidth<0.32in
  \PackageWarningNoLine{cover}{Spine is only \the\SpineWidth: shrink or drop
    the spine text (13pt type needs ~0.32in of spine)}
\fi
\newlength{\SheetWidth}  % 2 x 6.0" covers + 2 x 0.125" bleed + spine
\setlength{\SheetWidth}{\dimexpr 12.25in+\SpineWidth\relax}

\usepackage[paperwidth=\SheetWidth,paperheight=9.25in,margin=0in]{geometry}
\pagestyle{empty}
