\chapter{R7RS Quick Reference}
\index{R7RS!quick reference}

This appendix provides a concise reference to the R7RS-small standard. Use it as a lookup when you need to recall a syntax form, procedure signature, or which library something belongs to.

\section{Syntax Forms}

\begin{center}
\begin{tabular}{lp{8cm}}
\textbf{Form} & \textbf{Syntax} \\
\hline
\texttt{define} & \texttt{(define var expr)} or \texttt{(define (name params) body ...)} \\
\texttt{lambda} & \texttt{(lambda (params) body ...)} \\
\texttt{if} & \texttt{(if test consequent alternate)} \\
\texttt{cond} & \texttt{(cond (test expr ...) ... (else expr ...))} \\
\texttt{case} & \texttt{(case key ((datum ...) expr ...) ... (else expr ...))} \\
\texttt{and} & \texttt{(and test ...)} \\
\texttt{or} & \texttt{(or test ...)} \\
\texttt{when} & \texttt{(when test body ...)} \\
\texttt{unless} & \texttt{(unless test body ...)} \\
\texttt{let} & \texttt{(let ((var expr) ...) body ...)} \\
\texttt{let*} & \texttt{(let* ((var expr) ...) body ...)} \\
\texttt{letrec} & \texttt{(letrec ((var expr) ...) body ...)} \\
\texttt{letrec*} & \texttt{(letrec* ((var expr) ...) body ...)} \\
\texttt{let-values} & \texttt{(let-values (((v ...) expr) ...) body ...)} \\
\texttt{do} & \texttt{(do ((var init step) ...) (test result) body ...)} \\
\texttt{case-lambda} & \texttt{(case-lambda (formals body ...) ...)} \\
\texttt{begin} & \texttt{(begin expr ...)} \\
\texttt{quote} & \texttt{'datum} or \texttt{(quote datum)} \\
\texttt{quasiquote}\index{quasiquote} & \texttt{`template} with \texttt{,expr} and \texttt{,@expr} \\
\texttt{set!} & \texttt{(set! var expr)} \\
\texttt{define-syntax} & \texttt{(define-syntax name (syntax-rules ...))} \\
\texttt{syntax-rules} & \texttt{(syntax-rules (literals) (pattern template) ...)} \\
\texttt{let-syntax} & \texttt{(let-syntax ((name transformer) ...) body ...)} \\
\texttt{letrec-syntax} & \texttt{(letrec-syntax ((name transformer) ...) body ...)} \\
\texttt{define-library} & \texttt{(define-library (name ...) declarations ...)} \\
\texttt{import} & \texttt{(import import-set ...)} \\
\texttt{guard} & \texttt{(guard (var (test expr ...) ...) body ...)} \\
\texttt{define-record-type} & \texttt{(define-record-type name ctor pred field ...)} \\
\texttt{delay}\index{delay} & \texttt{(delay expr)} \\
\texttt{delay-force} & \texttt{(delay-force expr)} \\
\texttt{parameterize}\index{parameterize} & \texttt{(parameterize ((param val) ...) body ...)} \\
\texttt{cond-expand} & \texttt{(cond-expand (feature-req body ...) ...)} \\
\end{tabular}
\end{center}
